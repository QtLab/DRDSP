35  for(size_t i=0;i<data.count;++i) {
36  A1.block(i*dim,0,dim,ldim) = family.ComputeLinear( data.points[i] );
37  }
38  for(size_t i=0;i<data.count;++i) {
39  y1.segment(i*dim,dim) = data.vectors[i] - family.ComputeTranslation( data.points[i] );
40  }
41  for(size_t i=0;i<data.count;++i) {
42  A2.block(i*dim2,0,dim2,ldim) = family.ComputeLinearDerivative( data.points[i] );
43  }
44  for(size_t i=0;i<data.count;++i) {
45  y2.segment(i*dim2,dim2) = Vectorize( data.derivatives[i] - family.ComputeTranslationDerivative( data.points[i] ) );
46  }

48  auto A = A1.transpose() * A1 * (fitWeight[0]/data.scales[0]) + A2.transpose() * A2 * (fitWeight[1]/data.scales[1]);
49  auto b = A1.transpose() * y1 * (fitWeight[0]/data.scales[0]) + A2.transpose() * y2 * (fitWeight[1]/data.scales[1]);
