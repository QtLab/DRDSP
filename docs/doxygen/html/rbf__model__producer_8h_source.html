25  double ComputeTotalCost( const RBFModel<F>& model, const ReducedData& data ) const {
26  double S1 = 0.0;
27  for(uint32_t i=0;i<data.count;++i) {
28  S1 += ( model(data.points[i]) - data.vectors[i] ).squaredNorm();
29  }
30  S1 /= data.count;
31  double S2 = 0.0;
32  for(uint32_t i=0;i<data.count;++i) {
33  S2 += ( model.Partials(data.points[i]) - data.derivatives[i] ).squaredNorm();
34  }
35  S2 /= data.count;
36  return (fitWeight[0]/data.scales[0]) * S1 + (fitWeight[1]/data.scales[1]) * S2;
37  }

50  MatrixXd y1, y2, A1, A2, z1, z2, z, P, PI;
51  uint32_t m = data.dimension + model.numRBFs;
52  VectorXd temp;
53 
54  A1.setZero(m,data.count);
55  y1.setZero(data.dimension,data.count);
56  y2.setZero(data.dimension,data.count*data.dimension);
57  A2.setIdentity(m,data.count*data.dimension);
58 
59  for(uint32_t j=0;j<data.count;++j) {
60  for(uint32_t i=0;i<data.dimension;++i) {
61  A1(i,j) = data.points[j](i);
62  y1(i,j) = data.vectors[j](i);
63  for(uint32_t k=0;k<data.dimension;++k)
64  y2(i,data.dimension*j+k) = data.derivatives[j](i,k);
65  }
66  for(uint32_t i=0;i<model.numRBFs;++i) {
67  A1(i+data.dimension,j) = model.rbfs[i](data.points[j]);
68  temp = model.rbfs[i].Derivative(data.points[j]);
69  for(uint32_t k=0;k<data.dimension;++k)
70  A2(i+data.dimension,data.dimension*j+k) = temp(k);
71  }
72  }
73 
74  P = (A1 * A1.transpose()) * (fitWeight[0]/data.scales[0]) + (A2 * A2.transpose()) * (fitWeight[1]/data.scales[1]);
75 
76  FullPivLU<MatrixXd> lu(P);
77  PI = lu.inverse();
78 
79  z1 = y1 * A1.transpose() * PI;
80  z2 = y2 * A2.transpose() * PI;
