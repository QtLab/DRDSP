81  VectorXd trans, RHS;

85  N * data.reducedData[0].dimension * (1 + data.reducedData[0].dimension),
86  family.paramDim * ( parameters[0].size() + 1 )
87  );
88  RHS.setZero( LHS.rows() );
89 
90  int64_t k1 = 0, k2 = 0;
91  for(uint32_t i=0;i<data.numParameters;++i) {
92  D = ComputeD( parameters[i], family.paramDim );
93 
94  double scale = std::sqrt( fitWeight[0] / data.reducedData[i].scales[0] );
95  double scaleD = std::sqrt( fitWeight[1] / data.reducedData[i].scales[1] );
96 
97  for(uint32_t j=0;j<data.reducedData[i].count;++j) {
98  const VectorXd& x = data.reducedData[i].points[j];
99  linear = family.ComputeLinear( x );
100  trans = family.ComputeTranslation( x );
101  const VectorXd& V = data.reducedData[i].vectors[j];
102  LHS.block(k1,0,linear.rows(),D.cols()) = linear * D * scale;
103  RHS.segment(k2,V.size()) = (V - trans) * scale;
104  k1 += linear.rows();
105  k2 += V.size();
106  }
107  for(uint32_t j=0;j<data.reducedData[i].count;++j) {
108  const VectorXd& x = data.reducedData[i].points[j];
109  linearD = family.ComputeLinearDerivative( x );
110  transD = family.ComputeTranslationDerivative( x );
111  const MatrixXd& dV = data.reducedData[i].derivatives[j];
112  LHS.block(k1,0,linearD.rows(),D.cols()) = linearD * D * scaleD;
113  RHS.segment(k2,dV.size()) = Vectorize(dV - transD) * scaleD;
114  k1 += linearD.rows();
115  k2 += dV.size();
116  }
117  }
